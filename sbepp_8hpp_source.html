466 template<typename T, endian E, typename Byte>
467 SBEPP_CPP20_CONSTEXPR T get_primitive(const Byte* ptr)
468 {
469 #if SBEPP_HAS_BITCAST
470  std::array<Byte, sizeof(T)> arr;
471  if((E == endian::native) || (sizeof(T) == 1))
472  {
473  std::copy(ptr, ptr + sizeof(T), std::begin(arr));
474  }
475  else
476  {
477  std::reverse_copy(ptr, ptr + sizeof(T), std::begin(arr));
478  }
479  return std::bit_cast<T>(arr);
480 #else
481  // old compilers don't optimize `std::copy` approach good enough, that's
482  // why explicit `std::memcpy` call is required
483  T res;
484  std::memcpy(&res, ptr, sizeof(T));
485  if((E == endian::native) || (sizeof(T) == 1))
486  {
487  return res;
488  }
489  else
490  {
491  return static_cast<T>(byteswap(to_unsigned(res)));
492  }
493 #endif
494 }

496 template<endian E, typename T, typename Byte>
497 SBEPP_CPP20_CONSTEXPR void set_primitive(Byte* ptr, T value)
498 {
499 #if SBEPP_HAS_BITCAST
500  auto arr = std::bit_cast<std::array<Byte, sizeof(T)>>(value);
501  if((E == endian::native) || (sizeof(T) == 1))
502  {
503  std::copy(std::begin(arr), std::end(arr), ptr);
504  }
505  else
506  {
507  std::reverse_copy(std::begin(arr), std::end(arr), ptr);
508  }
509 #else
510  // old compilers don't optimize `std::copy` approach good enough, that's
511  // why explicit `std::memcpy` call is required
512  if((E != endian::native) && (sizeof(T) != 1))
513  {
514  value = static_cast<T>(byteswap(to_unsigned(value)));
515  }
516  std::memcpy(ptr, &value, sizeof(T));
517 #endif
518 }
