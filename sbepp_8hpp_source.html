514template<typename T, endian E, typename Byte>
515SBEPP_CPP20_CONSTEXPR T get_primitive(const Byte* ptr)
516{
517#if SBEPP_HAS_BITCAST
518 std::array<Byte, sizeof(T)> arr;
519 if((E == endian::native) || (sizeof(T) == 1))
520 {
521 std::copy(ptr, ptr + sizeof(T), std::begin(arr));
522 }
523 else
524 {
525 std::reverse_copy(ptr, ptr + sizeof(T), std::begin(arr));
526 }
527 return std::bit_cast<T>(arr);
528#else
529 // old compilers don't optimize `std::copy` approach good enough, that's
530 // why explicit `std::memcpy` call is required
531 T res;
532 std::memcpy(&res, ptr, sizeof(T));
533 if((E == endian::native) || (sizeof(T) == 1))
534 {
535 return res;
536 }
537 else
538 {
539 return from_unsigned<T>{}(byteswap(to_unsigned(res)));
540 }
541#endif
542}
543
544template<endian E, typename T, typename Byte>
545SBEPP_CPP20_CONSTEXPR void set_primitive(Byte* ptr, T value)
546{
547#if SBEPP_HAS_BITCAST
548 auto arr = std::bit_cast<std::array<Byte, sizeof(T)>>(value);
549 if((E == endian::native) || (sizeof(T) == 1))
550 {
551 std::copy(std::begin(arr), std::end(arr), ptr);
552 }
553 else
554 {
555 std::reverse_copy(std::begin(arr), std::end(arr), ptr);
556 }
557#else
558 // old compilers don't optimize `std::copy` approach good enough, that's
559 // why explicit `std::memcpy` call is required
560 if((E != endian::native) && (sizeof(T) != 1))
561 {
562 value = from_unsigned<T>{}(byteswap(to_unsigned(value)));
563 }
564 std::memcpy(ptr, &value, sizeof(T));
565#endif
566}
