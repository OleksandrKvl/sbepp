530template<typename T, endian E, typename Byte>
531SBEPP_CPP20_CONSTEXPR T get_primitive(const Byte* ptr)
532{
533#if SBEPP_HAS_BITCAST
534 std::array<Byte, sizeof(T)> arr;
535 if((E == endian::native) || (sizeof(T) == 1))
536 {
537 std::copy(ptr, ptr + sizeof(T), std::begin(arr));
538 }
539 else
540 {
541 std::reverse_copy(ptr, ptr + sizeof(T), std::begin(arr));
542 }
543 return std::bit_cast<T>(arr);
544#else
545 // old compilers don't optimize `std::copy` approach good enough, that's
546 // why explicit `std::memcpy` call is required
547 T res;
548 std::memcpy(&res, ptr, sizeof(T));
549 if((E == endian::native) || (sizeof(T) == 1))
550 {
551 return res;
552 }
553 else
554 {
555 return from_unsigned<T>{}(byteswap(to_unsigned(res)));
556 }
557#endif
558}
559
560template<endian E, typename T, typename Byte>
561SBEPP_CPP20_CONSTEXPR void set_primitive(Byte* ptr, T value)
562{
563#if SBEPP_HAS_BITCAST
564 auto arr = std::bit_cast<std::array<Byte, sizeof(T)>>(value);
565 if((E == endian::native) || (sizeof(T) == 1))
566 {
567 std::copy(std::begin(arr), std::end(arr), ptr);
568 }
569 else
570 {
571 std::reverse_copy(std::begin(arr), std::end(arr), ptr);
572 }
573#else
574 // old compilers don't optimize `std::copy` approach good enough, that's
575 // why explicit `std::memcpy` call is required
576 if((E != endian::native) && (sizeof(T) != 1))
577 {
578 value = from_unsigned<T>{}(byteswap(to_unsigned(value)));
579 }
580 std::memcpy(ptr, &value, sizeof(T));
581#endif
582}
